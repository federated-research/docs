import { useState } from "react";
import { LikeC4View } from "../../likec4_react/likec4.generated";

export default function Diagram({
  alt_image = "/images/likec4_index.png",
  alt_text = "Architecture diagram",
  viewId = "index",
}) {
  const [interactive, setInteractive] = useState(false);

if (!interactive) {
return (

<div style={{ width: "100%" }}>
 <button
 type="button"
 onClick={() => setInteractive(true)}
 style={{
 border: "none",
 padding: 0,
 background: "none",
 width: "100%",
 display: "block",
 cursor: "pointer",
 }}
 aria-label={alt_text}
 >
<img
src={alt_image}
alt={alt_text}
style={{ width: "100%", display: "block" }}
/>
</button>
</div>
);
}

return (

<div style={{ width: "100%" }}>
<LikeC4View viewId={viewId} />
</div>
);
}
